Example 1

01 private final ContainerConfig config;
02 public static final String PROXY_PATH_PARAM="gadgets.uri.proxy.path";
03 static final String CHAINED_PARAMS_TOKEN="%chained_params%";
04 Uri uriIn;
05 String container=uriIn.getQueryParameter(Param.CONTAINER.getKey());
06 if(container == null)
07 {
08 container=uriIn.getQueryParameter(Param.SYND.getKey());
09 }
10 String containerStr=Param.CONTAINER.getKey() + '=';
11 String path=uriIn.getPath();
12 boolean doDecode=(!path.contains(containerStr));
13 if(doDecode)
14 {
15 path=Utf8UrlCoder.decode(path);
16 }
17 int start=path.indexOf(containerStr);
18 if(start > 0)
19 {
20 start+=containerStr.length();
21 int end=path.indexOf('&',start);
22 if(end < start)
23 {
24 end=path.indexOf('/',start);
25 }
26 if(end > start)
27 {
28 container=path.substring(start,end);
29 if(container != null)
30 {
31 String proxyPath=config.getString(container,PROXY_PATH_PARAM);
32 //proxyPath -> the String to parse, may be null
33 //CHAINED_PARAMS_TOKEN -> String containing the String to be used as a delimiter, null splits on whitespace
34 StringUtils.splitByWholeSeparatorPreserveAllTokens(proxyPath,CHAINED_PARAMS_TOKEN);
35 }
36 }
37 }
