Example 1

01 /**
02 * Number of base package levels that is shown as a single node in the tree at the
03 * root level. Should be at least 2, because most packages start with "io.trumpet" or "org.apache" or "com.foo".
04 */
05 public static final int PRESERVE_PACKAGES=2;
06 final String loggerName;
07 String[] pieces=StringUtils.split(loggerName,".");
08 //pieces -> the array of values to join together, may be null
09 //"." -> the separator character to use, null treated as ""
10 //0 -> the first index to start joining from. It is an error to pass in an end index past the end of the array
11 //PRESERVE_PACKAGES -> the index to stop joining from (exclusive). It is an error to pass in an end index past the end of the array
12 StringUtils.join(pieces,".",0,PRESERVE_PACKAGES);
