Example 2

01 JarFile jar;
02 Enumeration<JarEntry> entries=jar.entries();
03 while(entries.hasMoreElements())
04 {
05 JarEntry element=entries.nextElement();
06 String name=element.getName();
07 if(name.toLowerCase().endsWith(".class"))
08 {
09 String className=StringUtils.replace(name,"/",".");
10 //className -> the source String to search, may be null
11 //".class" -> the String to search for and remove, may be null
12 className=StringUtils.removeEnd(className,".class");
13 }
14 }
