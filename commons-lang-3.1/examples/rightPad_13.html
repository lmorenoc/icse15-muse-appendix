01 private final long expectedByteCount;
02 private long snapshotBytes;
03 private long snapshotStart=System.currentTimeMillis();
04 private long bytesPerSecond;
05 final int bytesWritten;
06 final long byteCount=getByteCount();
07 final double progress=((double)byteCount / (double)expectedByteCount) * 100.0;
08 snapshotBytes+=bytesWritten;
09 if(System.currentTimeMillis() - snapshotStart >= 1000)
10 {
11 bytesPerSecond=snapshotBytes;
12 }
