Example 6

01 /**
02 * Maximum length to show.
03 */
04 public static final int MAX=100;
05 final String text;
06 String result;
07 if(text.length() < TextDecor.MAX)
08 {
09 result=text;
10 }
11 final int skip=text.length() - TextDecor.MAX;
12 final StringBuilder output=new StringBuilder().append(text.substring(0,(text.length() - skip) / 2)).append("..").append(skip).append("..");
13 result=output.toString();
14 //result -> String to escape values in, may be null
15 StringEscapeUtils.escapeJava(result);
