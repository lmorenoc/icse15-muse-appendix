01 File basedir;
02 try
03 {
04 Collection<File> files=FileUtils.listFiles(basedir,FileFilterUtils.or(FileFilterUtils.suffixFileFilter("jar"),FileFilterUtils.suffixFileFilter("xml"),FileFilterUtils.suffixFileFilter("zip")),FileFilterUtils.or(FileFilterUtils.nameFileFilter("lib"),FileFilterUtils.nameFileFilter("pom"),FileFilterUtils.nameFileFilter("assembly")));
05 if(files.isEmpty())
06 {

08 }
09 for(File file : files)
10 {
11 try
12 {
13 String base=FilenameUtils.getBaseName(file.getName().replaceAll("_","-"));
14 if("xml".equals(FilenameUtils.getExtension(file.getName())))
15 {
16 continue;
17 }
18 IOFileFilter filter=new WildcardFileFilter(base + ".pom");
19 Collection<File> poms=FileUtils.listFiles(basedir,filter,TrueFileFilter.INSTANCE);
20 if(poms.size() != 1)
21 {
22 continue;
23 int index=base.indexOf("-");
24 String prefix=base.substring(0,index);
25 }
26 }
27 //prefix -> the filename prefix
28 FileFilterUtils.prefixFileFilter(prefix);
29 }
