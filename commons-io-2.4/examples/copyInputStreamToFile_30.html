09 for(Map.Entry<String,AntPathMatcherFilter> filterEntry : content.entrySet())
10 {
11 AntPathMatcherFilter filter=filterEntry.getValue();
12 if(filter.accept(file.getName()))
13 {
14 continue outer;
15 }
16 continue outer;
17 }
18 ZipFile zip=new ZipFile(file);
19 Enumeration<? extends ZipEntry> entries=zip.entries();
20 while(entries.hasMoreElements())
21 {
22 ZipEntry entry=entries.nextElement();
23 if(entry.isDirectory())
24 {
25 continue;
26 }
27 for(Map.Entry<String,AntPathMatcherFilter> filterEntry : content.entrySet())
28 {
29 AntPathMatcherFilter filter=filterEntry.getValue();
30 if(filter.accept(entry.getName()))
31 {
32 File target=FileUtils.getFile(to,ArchiveUtils.rewritePath(entry.getName(),filterEntry.getKey()));
33 //zip.getInputStream(entry) -> the <code>InputStream</code> to copy bytes from, must not be null
34 //target -> the non-directory <code>File</code> to write bytes to (possibly overwriting), must not be null
35 FileUtils.copyInputStreamToFile(zip.getInputStream(entry),target);
36 }
37 }
38 }
39 }
40 }
