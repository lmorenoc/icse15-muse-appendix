Example 6

01 String newValue;
02 String oldValue;
03 String path=newValue;
04 String filename;
05 if("".equals(FilenameUtils.getExtension(newValue)))
06 {
07 filename=FilenameUtils.getName(oldValue);
08 }
09 path=FilenameUtils.getPath(newValue);
10 filename=FilenameUtils.getName(newValue);
11 Pattern p=Pattern.compile("(.*?)#(.*?)(?:$|(/.*))");
12 Matcher m=p.matcher(path);
13 if(m.find())
14 {
15 String prefix=m.group(1);
16 String dir=m.group(2);
17 String suffix=m.group(3);
18 int index=oldValue.indexOf(dir);
19 int length=dir.length();
20 String remainder=FilenameUtils.getPath(oldValue.substring(index + length));
21 if(prefix != null && !prefix.equals(""))
22 {
23 path=prefix + remainder;
24 }
25 }
26 path=remainder;
27 if(suffix != null && !suffix.equals(""))
28 {
29 path+=suffix;
30 }
31 //path+"/"+filename -> the filename to normalize, null returns null
32 FilenameUtils.normalize(path + "/" + filename);
