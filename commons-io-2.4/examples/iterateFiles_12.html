1 public static Pattern FILE_TEMPLATE_PATTERN=Pattern.compile(".*(\\.template)\\..*");
2 File root;
3 //root -> the directory to search in
4 //newRegexFileFilter(FILE_TEMPLATE_PATTERN) -> filter to apply when finding files.
5 //TrueFileFilter.TRUE -> optional filter to apply when finding subdirectories. If this parameter is null , subdirectories will not be included in the search. Use TrueFileFilter.INSTANCE to match all directories.
6 FileUtils.iterateFiles(root,new RegexFileFilter(FILE_TEMPLATE_PATTERN),TrueFileFilter.TRUE);
