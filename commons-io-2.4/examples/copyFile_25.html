Example 3

01 /**
02 * Karma-junit-reporter results file. Setting this location will export the results file to the specified reportsDirectory. For this
03 * to function, the karma-junit-reporter plugin must be included in the karma configuration file and the reportsDirectory
04 * must be available for writing. The junit reporter plugin's outputFile property should also be configured in karma.conf.js to
05 * be pointing to the same directory as this property, ensuring that the file is picked up.
06 */
07 @Parameter(property="junitReportFile",required=false) private File junitReportFile;
08 /**
09 * Base directory where all Karma reports are written to. This occurs in the post execution step where the file
10 * specified in the junitReportFile is copied to this directory.
11 */
12 @Parameter(defaultValue="${project.build.directory}/karma-reports",required=false) private File reportsDirectory;
13 //junitReportFile -> an existing file to copy, must not be null
14 //newFile(reportsDirectory,junitReportFile.getName()) -> the new file, must not be null
15 FileUtils.copyFile(junitReportFile,new File(reportsDirectory,junitReportFile.getName()));

01 private Set<File> files=new HashSet<File>();
02 private Map<String,AntPathMatcherFilter> content=new HashMap<String,AntPathMatcherFilter>();
03 File to;
04 for(File file : files)
05 {
06 String ext=FilenameUtils.getExtension(file.getName()).toUpperCase();
07 if(!"ZIP".equals(ext))
08 {
09 for(Map.Entry<String,AntPathMatcherFilter> filterEntry : content.entrySet())
10 {
11 AntPathMatcherFilter filter=filterEntry.getValue();
12 if(filter.accept(file.getName()))
13 {
14 File target=FileUtils.getFile(to,ArchiveUtils.rewritePath(file.getName(),filterEntry.getKey()));
15 //file -> an existing file to copy, must not be null
16 //target -> the new file, must not be null
17 FileUtils.copyFile(file,target);
18 }
19 }
20 }
21 }
